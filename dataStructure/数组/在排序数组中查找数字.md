## 题目

统计一个数字在排序数组中出现的次数. 


## 思路

本道题有好几种解法

1. 直接遍历数组,判断前后的值是否相同,找到元素开始位置和结束位置,时间复杂度O(n)

```js
function findNumTimes(arr, k) {
  if (Array.isArray(arr)) {
    let firstIndex = arr.indexOf(k)

    if (firstIndex === -1) {
      return 0
    }

    let lastIndex = arr.lastIndexOf(k)

    if (firstIndex !== lastIndex) {
      return lastIndex - firstIndex + 1
    } else {
      return 1
    }
  }
}

时间复杂度O(n)
```

2. 使用二分查找分别找到第一个目标值出现的位置和最后一个位置,时间复杂度O(logn)

在排序数组中找元素,首先考虑使用二分查找！！！！！

下面是使用二分查找在数组中寻找某个数

```js
function getNumOfK(nums, k) {
  //  防止k 无输入
  if (nums.length === 0 || !k) {
    return 0
  }

  let firstK = findFirstK(nums, k)
  let lastK = findLastK(nums, k)
  //  有可能找不到
  if (firstK === -1 || lastK === -1) {
    return 0
  } else {
    //  相等的情况同样适用
    return lastK - firstK + 1
  }
}

function findFirstK(nums, k) {
  let left = 0
  let right = nums.length - 1

  while(left <= right) {
      let mid = Math.floor((left + right) / 2)

    if (nums[mid] < k) {
      left = mid + 1
    } else if (nums[mid] > k) {
      right = mid - 1
    } else {
      if (mid === 0 || nums[mid - 1] !== k) {
        return mid
      } else {
        right = mid - 1
      }
    }
  }

  return -1
}

function findLastK(nums, k) {
  let left = 0
  let right = nums.length - 1


  while(left <= right) {
      let mid = Math.floor((left + right) / 2)

    if (nums[mid] < k) {
      left = mid + 1
    } else if (nums[mid] > k) {
      right = mid - 1
    } else {
      if (mid === nums.length - 1 || nums[mid + 1] !== k) {
        return mid
      } else {
        left = mid + 1
      }
    }
  }

  return -1
}















function getNumberOfK(nums, k) {
  if (nums.length === 0) {
    return 0;
  }
  let first = findFirstK(nums, k);
  let last = findLastK(nums, k);
  if (first === -1 || last === -1) {
    return 0;
  } else {
    return last - first + 1;
  }
}
//  查找第一个k在排序数组中的位置
function findFirstK(nums, k) {
  let left = 0;
  let right = nums.length - 1;
  while (left <= right) {
    let mid = Math.floor((left + right) / 2);
    if (nums[mid] < k) {
      left = mid + 1;
    } else if (nums[mid] > k) {
      right = mid - 1;
    } else {  //  刚好相等的情况
      if (mid === 0 || nums[mid - 1] !== k) {
        return mid;
      } else {
        right = mid - 1;
      }
    }
  }
  return -1;
}
//  查找最后一个k在排序数组中的位置
function findLastK(nums, k) {
  let left = 0;
  let right = nums.length - 1;
  while (left <= right) {
    let mid = Math.floor((left + right) / 2);
    if (nums[mid] < k) {
      left = mid + 1;
    } else if (nums[mid] > k) {
      right = mid - 1;
    } else {  //  刚好相等的情况
      if (mid === nums.length - 1 || nums[mid + 1] !== k) {
        return mid;
      } else {
        left = mid + 1;
      }
    }
  }
  return -1;
}

时间复杂度O(logn)
```
