
`0,1,...,n-1`这n个数字排成一个圆圈，从数字0开始，每次从这个圆圈里删除第m个数字. 求出这个圆圈里剩下的最后一个数字. 

其实这就是著名的约瑟夫环问题，

```js

/**
 * n 个数
 * m 第m个数被删除
 */
let lastRemaining = (n, m) => {

  //  创建0-n-1 的数组
  let nums = Array.from({length: n}, (_, i) => i)
  //  要删除的下标
  let index = 0
  while(nums.length > 1) {
    //  因为可能需要多圈之后才能剩下最后一个数
    index = (index + m - 1) % nums.length 
    nums.splice(index, 1)
  }

  return nums[0]

}


//    tip
// 使用箭头函数作为映射函数去操作多个元素
Array.from([1, 2, 3], (x) => x + x);
// [2, 4, 6]

// 生成一个数字序列. 因为数组在每个位置都使用 `undefined` 初始化，下面的 `v` 值将是 `undefined`
Array.from({ length: 5 }, (v, i) => i);
// [0, 1, 2, 3, 4]

const set = new Set(["foo", "bar", "baz", "foo"]);
Array.from(set);
// [ "foo", "bar", "baz" ]

取余(%）运算符返回左侧操作数除以右侧操作数的余数. 它总是与被除数的符号保持一致. 
console.log(13 % 5);
// Expected output: 3

console.log(-13 % 5);
// Expected output: -3

console.log(4 % 2);
// Expected output: 0

console.log(-4 % 2);
// Expected output: -0
```
