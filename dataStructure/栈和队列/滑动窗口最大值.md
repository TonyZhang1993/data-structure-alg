## 题目 - TBD

给定一个数组 nums，有一个大小为 k 的滑动窗口从数组的最左侧移动到数组的最右侧。你只可以看到在滑动窗口 k 内的数字。滑动窗口每次只向右移动一位。 返回滑动窗口最大值。

```
输入: nums = [1,3,-1,-3,5,3,6,7], 和 k = 3
输出: [3,3,5,5,6,7] 
解释: 
  滑动窗口的位置               最大值
---------------              -----
[1  3  -1] -3  5  3  6  7       3
 1 [3  -1  -3] 5  3  6  7       3
 1  3 [-1  -3  5] 3  6  7       5
 1  3  -1 [-3  5  3] 6  7       5
 1  3  -1  -3 [5  3  6] 7       6
 1  3  -1  -3  5 [3  6  7]      7

```

```js
var maxSlidingWindow = function (nums, k) {
  const window = []; // 创建空数组，用于表示当前滑动窗口
  const result = []; // 创建空数组，用于存储所有滑动窗口的最大值
  for (let i = 0; i < nums.length; i++) { // 遍历数组
    if (i - window[0] > k - 1) { // 如果滑动窗口中最左端元素已经超出了当前窗口的范围
      window.shift(); // 从滑动窗口中删除最左端的元素
    }
    let j = window.length - 1; // 设置指针 j 指向滑动窗口中最右端的元素
    while (j >= 0 && nums[window[j]] <= nums[i]) { // 当滑动窗口中存在比当前元素小的元素时
      j--; // 指针左移，继续查找
      window.pop(); // 从滑动窗口中删除该元素及其右侧的所有元素，保持滑动窗口的单调递减性质
    }
    window.push(i); // 将当前元素的索引加入滑动窗口
    if (i >= k - 1) { // 如果当前元素的索引大于等于 k-1，说明已经有足够的元素填满了一个大小为 k 的窗口
      result.push(nums[window[0]]); // 将滑动窗口中的最左端元素加入结果数组，即为当前窗口的最大值
    }
  }
  return result; // 返回结果数组，其中存储了所有滑动窗口的最大值
};
```
