## 面试真题: 

删除字符串中出现次数 >= 2 次的相邻字符

```
输入: "abbbaca"
输出: "ca"
解释: "abbbaca" => "aaca"=>"ca"

```

```js
let removeDuplicated = (s) => {
  let stack = []
  //  栈顶元素 / 遍历字符串s 的字符 / 字符串s的下标
  let top, next, i = 0

  while(i < s.length) {
    //  栈顶元素
    top = stack[stack.length - 1]
    //  当前字符串s 的字符
    next = s[i]
    //  若相等 - 推出栈顶元素
    if (top === next) {
      stack.pop()
      //  跳过重复的字符
      while(top === s[i]) i++
    } else {
      stack.push(s[i])
      i++
    }
  }

  return stack.join('')
}

```
