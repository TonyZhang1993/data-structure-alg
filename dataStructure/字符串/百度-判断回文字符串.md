## 题目

百度: 实现一个函数，判断输入是不是回文字符串

```js
//  api
const isPlalindrome = (strs）=> {
  if (typeof strs !== 'string') return false
  return strs.split('').reverse().join('') === strs
}

//  双指针
const isPlalindrome = (strs）=> {
  if (typeof strs !== 'string') return false
  
  let i=0, j=strs.length-1
  while(i<j) {
    if (strs.charAt(i) === strs.charAt(j)) {
      i++
      j--
    } else {
      return false
    }
  }

  return true
}
```
