```js
function reverseSentence(str) {
  if (!str) return ''
  return str.split(' ').reverse().join(' ')
}
```

```js
  if (!str) return ''
  if (num < 0) return ''

  let len = str.length

  if (num >= len) {
    return str
  } else {
    // const tmp = str.slice(0, num+1)
```
