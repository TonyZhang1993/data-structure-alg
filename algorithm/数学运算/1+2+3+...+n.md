求1+2+3+...+n,要求不能使用乘除法, `for, while, if, else, switch, case`等关键字及条件判断语句`(A?B:C)`. 

```js
function addN(n) {
  let count = 0
  for (let i=1; i<=n; i++) {
    count+=i
  }

  return count
}
//  公式 (1+n)n / 2 = (n + n^2) / 2
function addN(n) {
  //  位运算代替除法 - 将 结果 除以 2
  return (Math.pow(n, 2) + n) >> 1
}


let num = 16;
num = (num >> 2) & 3; // 将 num 除以 4
```
