```js

function countOnesInRange(start, end) {
  let count = 0;
  for (let num = start; num <= end; num++) {
    count += countOnes(num);
  }
  return count;
}
//  计算一个整数中数字 1 出现的次数
function countOnes(num) {
  let count = 0;
  while (num > 0) {
    if (num % 10 === 1) {
      count++;
    }
    num = Math.floor(num / 10);
  }
  return count;
}
```
