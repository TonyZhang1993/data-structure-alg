统计一个数字在排序数组中出现的次数. 

## 思路
在排序数组中找元素，首先考虑使用二分查找



```js
//  方法一
function findTimes(arr, num) {
  let first = arr.indexOf(num)

  if (first === -1) return 0

  let last = arr.lastIndexOf(num)

  return last - first + 1
}
O(n)

//  方法二
function countOccurrences(arr, target) {
  let left = 0;
  let right = arr.length - 1;
  let firstOccurrence = findFirstOccurrence(arr, target);
  let lastOccurrence = findLastOccurrence(arr, target);

  if (firstOccurrence !== -1 && lastOccurrence !== -1) {
    return lastOccurrence - firstOccurrence + 1;
  }

  return 0;
}

function findFirstOccurrence(arr, target) {
  let left = 0;
  let right = arr.length - 1;
  //  注意这个玩意儿的定义
  let firstOccurrence = -1;

  while (left <= right) {
    let mid = Math.floor((left + right) / 2);

    if (arr[mid] === target) {
      firstOccurrence = mid;
      //  key!!!
      right = mid - 1; // 继续查找左侧部分
    } else if (arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }
  //  返回该东东
  return firstOccurrence;
}

function findLastOccurrence(arr, target) {
  let left = 0;
  let right = arr.length - 1;
  let lastOccurrence = -1;

  while (left <= right) {
    let mid = Math.floor((left + right) / 2);

    if (arr[mid] === target) {
      lastOccurrence = mid;
      //  key!!!
      left = mid + 1; // 继续查找右侧部分
    } else if (arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }

  return lastOccurrence;
}

时间复杂度 O(log n)
```
